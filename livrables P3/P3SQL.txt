SELECT Pays, Annee, SUM(dispo_prot)/1000 AS Dispo_Prot
FROM dispo_alim 
WHERE Annee="2013" 
GROUP BY Pays, Annee
ORDER BY Dispo_Prot DESC
LIMIT 10;

FROM dispo_alim 
WHERE Annee="2013" 
GROUP BY Pays, Annee
ORDER BY Dispo_Kcal DESC
LIMIT 10;

SELECT sous_nutrition.pays,(sous_nutrition.nb_personnes)/(population.Population)*100 AS Ratio
FROM sous_nutrition JOIN population
ON sous_nutrition.code_pays=population."Code Pays"
WHERE (sous_nutrition.Annee="2012-2014" AND population.Annee="2013")
ORDER BY (sous_nutrition.nb_personnes)/(population.Population)*100 DESC
LIMIT 10;
